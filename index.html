Sample

Usage


import akka.actor._				  
val sys = ActorSystem()
val worker = sys.actorOf(Props[Worker])				  
		      
worker ! DoStuff("details")
val question: Future[Question] = worker ? 42

Sample

Defining actors


import akka.actor._
				  
case class DoStuff(s: String)

class Worker extends Actor {
  case DoStuff(s) => println(s)
  case n: Int => sender ! "You'll need a bigger computer for that"
}

Routing


import akka.routing.RoundRobin			
val worker = sys.actorOf(
  Props[Worker] withRouter RoundRobin(5),
  "myWorker" 
)

Config

Exact same code!


akka.actor.deployment {
  /myWorker {
    router = random-router
    nr-of-instances = 17
  }
}
